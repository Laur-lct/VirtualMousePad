# function takes original image as input and outputs
# augmented larger image, where original image is shifted/scaled
# mean and standart deviation values used to create random noise on resulting image
def generate_augmented_image(img, mean, std, scale, shift):
    res = np.zeros(sizeDest, np.uint8)
    # add noise
    cv2.randn(res, mean, std)
    
    newCenter = [centerDest[0]+shift[0], centerDest[1]+shift[1]]
    # calculate 3 points for translation matrix    
    ptsDest = np.float32([
        [newCenter[0] - centerOrig[0]*scale, newCenter[1] - centerOrig[1]*scale],
        [newCenter[0] - centerOrig[0]*scale, newCenter[1] + centerOrig[1]*scale - 1],
        newCenter])
    # create translation matrix
    mat = cv2.getAffineTransform(ptsOrig, ptsDest)
    cv2.warpAffine(src=img, M=mat, dsize=(sizeDest[0], sizeDest[1]), dst=res, borderMode=cv2.BORDER_TRANSPARENT)
    return res


# out of one original image, this function creates a set of scalesCount*shiftsCount augmented images
# with random shifts and scales, as specified in shiftRange and scaleRange settings
# resulting images saved to the augmented training/test dataset
# output is incremental number of stored train images and decremental number of reamaining test dataset places
def save_image_set(image, label, batchcount, testcount):
    for i in range(0, scalesCount):
        scale = random.uniform(scaleRange[0], scaleRange[1])
        for j in range(0, shiftsCount):
            shift = np.random.randint(shiftRange[0], shiftRange[1], 2)
            augmented = generate_augmented_image(image, mean, std, scale, shift)
            fname = outFileFormatPattern.format('C' if label == '0' else 'O', count, batchcount)
            destPath = datasetTrainPath
            labelsFile = labelsFileTrain
            # decide if this image goes to train or test set             
            if random.random() < testDatasetSize and testcount > 0:
                destPath = datasetTestPath
                labelsFile = labelsFileTest
                testcount -= 1
                
            # save the image and its label                
            cv2.imwrite(destPath + fname, augmented, [int(cv2.IMWRITE_JPEG_QUALITY), 100])
            labelsFile.write(fname + ' ' + label + os.linesep)
            batchcount += 1
    return batchcount, testcount

# open label files
labelsFileTrain = open(labelsFileTrainPath, 'a')
labelsFileTest = open(labelsFileTestPath, 'a')
count = 0
for (dirpath, dirnames, filenames) in os.walk(datasetInitialPath):
    if len(filenames) == 0:
        continue
    fnamesCount = len(filenames)
    folderFilesCount=0
    progress = hlp.ProgressBar(fnamesCount, '('+dirpath+')')
    testCount = (scalesCount * shiftsCount * 2 * fnamesCount * testDatasetSize)
    for imgfile in filenames:
        progress.animate(folderFilesCount)
        count += 1
        folderFilesCount+=1
        if '.jpg' not in imgfile:
            continue
        
        
        
        img = cv2.imread(dirpath + '/' + imgfile, cv2.IMREAD_GRAYSCALE)
        # create flipped image copy        
        imgFlipped = cv2.flip(img, 1)
        mean = np.mean(img)
        std = np.std(img)
        label = '0' if 'closed_eye' in imgfile else '1'

        batchCount = 0
        batchCount, testCount = save_image_set(img, label, 0, testCount)
        batchCount, testCount = save_image_set(imgFlipped, label, batchCount, testCount)

print ' *** '
print 'Train and test datasets generated.'        
labelsFileTrain.close()
labelsFileTest.close()
