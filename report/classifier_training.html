# helper function, performs testing of the network
# returns accuracy and test loss
def test_network(test_net, progress_bar, step):
    correct = 0
    lossSum = 0
    for test_it in range(100):
        test_net.forward()
        correct += sum(test_net.blobs['score'].data.argmax(1) == test_net.blobs['label'].data)
        lossSum += test_net.blobs['loss'].data
    acc = correct / 1e4 # todo remove constant
    progress_bar.text = ". Test accuracy: %.4f. Test loss: %.4f" %(acc, lossSum / 100)
    progress_bar.animate(step)
    return acc, lossSum / 100


# performs NN training, takes Caffe solver and number of training iterations as an input
# draws training chart when finished and returns the trained network
def train_network(solver, n_iter):
    train_loss = zeros(n_iter)
    test_acc = zeros(int(np.ceil(n_iter / test_interval)) +1)
    test_loss = zeros(len(test_acc))
    progress = hlp.ProgressBar(len(test_acc))
    
    for it in range(n_iter):
        solver.step(1)  # SGD by Caffe
        # store the train loss
        train_loss[it] = solver.net.blobs['loss'].data

        # run a full test every so often
        if it % test_interval == 0:
            step = it // test_interval
            test_acc[step], test_loss[step] = test_network(solver.test_nets[0], progress, step)
            
    test_acc[-1], test_loss[-1] = test_network(solver.test_nets[0], progress, len(test_acc)-1)
    
    # draw the chart
    _, ax1 = plt.subplots()
    ax2 = ax1.twinx()
    ax1.plot(arange(n_iter), train_loss)
    ax1.plot(test_interval * arange(len(test_loss)), test_loss, 'g')
    ax2.plot(test_interval * arange(len(test_acc)), test_acc, 'r')
    ax1.set_xlabel('iteration')
    ax1.set_ylabel('train & test loss')
    ax2.set_ylabel('test accuracy')
    ax2.set_title('Test Accuracy: {:.4f}'.format(test_acc[-1]))
    plt.show()

    return solver.test_nets[0]

# creates and saves the benchmark LeNet model
def lenet(isTrain, batch_size):
    # our version of LeNet adapted for 2 class output
    modelName = modelPath + 'lenet_' + ('train' if isTrain else 'test')
    n = caffe.NetSpec()
    lmdb = (datasetTrainPath if isTrain else datasetTestPath)
    n.data, n.label = L.Data(batch_size=batch_size, backend=P.Data.LMDB, source=lmdb,
                             transform_param=dict(mean_file=lmdb + '/mean.binaryproto', scale=1. / 255),
                             ntop=2)

    n.conv1 = L.Convolution(n.data, kernel_size=5, num_output=20, weight_filler=dict(type='xavier'))
    n.pool1 = L.Pooling(n.conv1, kernel_size=2, stride=2, pool=P.Pooling.MAX)
    n.conv2 = L.Convolution(n.pool1, kernel_size=5, num_output=50, weight_filler=dict(type='xavier'))
    n.pool2 = L.Pooling(n.conv2, kernel_size=2, stride=2, pool=P.Pooling.MAX)
    n.fc1 = L.InnerProduct(n.pool2, num_output=500, weight_filler=dict(type='xavier'))
    n.relu1 = L.ReLU(n.fc1, in_place=True)
    n.score = L.InnerProduct(n.relu1, num_output=2, weight_filler=dict(type='xavier'))
    n.loss = L.SoftmaxWithLoss(n.score, n.label)
    proto = n.to_proto()

    # store
    with open(modelName + '.prototxt', 'w') as f:
        f.write(str(proto))

    # visualize the network
    if isTrain:
        caffe.draw.draw_net_to_file(caffe_net=proto, filename=modelName + '.png', rankdir='LR')
        print "Banchmark LeNet:"
        with open(modelName + '.png', "rb") as imageFile:
            f = imageFile.read()
            b = bytearray(f)
            display(HTML("<img src='data:image/png;base64,{0}'/>".format(b64encode(b).decode('utf-8'))))

    return modelName + '.prototxt'

# creates and saves the classifier model
def model(isTrain, batch_size):
    modelName = modelPath + 'model_' + ('train' if isTrain else 'test')
    n = caffe.NetSpec()
    lmdb = (datasetTrainPath if isTrain else datasetTestPath)
    n.data, n.label = L.Data(batch_size=batch_size, backend=P.Data.LMDB, source=lmdb,
                             transform_param=dict(mean_file=lmdb + '/mean.binaryproto', scale=1. / 255),
                             ntop=2)

    n.conv1 = L.Convolution(n.data, kernel_size=6, num_output=45, stride=1, weight_filler=dict(type='xavier'))
    n.relu1 = L.ReLU(n.conv1, in_place=True)
    n.pool1 = L.Pooling(n.relu1, kernel_size=4, stride=2, pool=P.Pooling.MAX)
    
    n.conv2 = L.Convolution(n.pool1, kernel_size=3, stride=1, num_output=80, weight_filler=dict(type='xavier'))
    n.relu2 = L.ReLU(n.conv2, in_place=True)
    n.pool2 = L.Pooling(n.relu2, kernel_size=2, stride=2, pool=P.Pooling.MAX)
    
    n.fc1 = L.InnerProduct(n.pool2, num_output=600, weight_filler=dict(type='xavier'))
    n.relu3 = L.ReLU(n.fc1, in_place=True)
    
    n.score = L.InnerProduct(n.relu3, num_output=2, weight_filler=dict(type='xavier'))
    n.loss = L.SoftmaxWithLoss(n.score, n.label)
    proto = n.to_proto()

    # store
    with open(modelName + '.prototxt', 'w') as f:
        f.write(str(proto))

    # visualize the network
    if isTrain:
        caffe.draw.draw_net_to_file(caffe_net=proto, filename=modelName + '.png', rankdir='LR')
        print "Classifier Network:"
        with open(modelName + '.png', "rb") as imageFile:
            f = imageFile.read()
            b = bytearray(f)
            display(HTML("<img src='data:image/png;base64,{0}'/>".format(b64encode(b).decode('utf-8'))))

    return modelName + '.prototxt'

import copy
# shows confusion matrix and typical classification mistake examples
def confusion_and_mistakes(test_net):
    y_predicted = []
    y_true = []
    examples_false_positive = []
    examples_false_negative = []
    for test_it in range(100):
        test_net.forward()
        predicted_batch =test_net.blobs['score'].data.argmax(1)
        y_predicted.extend(predicted_batch)
        y_true.extend(test_net.blobs['label'].data)
        for i in range(len(predicted_batch)):
            if predicted_batch[i] == 0 and test_net.blobs['label'].data[i]==1 and len(examples_false_negative) < 10:
                examples_false_negative.append(copy.deepcopy(test_net.blobs['data'].data[i][0]))
            elif predicted_batch[i] == 1 and test_net.blobs['label'].data[i]==0 and len(examples_false_positive) < 10:
                examples_false_positive.append(copy.deepcopy(test_net.blobs['data'].data[i][0]))

    plt.figure()
    hlp.plot_confusion_matrix(confusion_matrix(y_true, y_predicted), ['closed', 'opened'])
    plt.show()  
    
    print "False positive examples:"
    f,ax = plt.subplots(1,len(examples_false_negative), figsize=(75,75))
    for i in range(len(examples_false_positive)):
        ax[i].axes.get_xaxis().set_visible(False)
        ax[i].axes.get_yaxis().set_visible(False)
        ax[i].imshow(examples_false_positive[i], cmap='gray')
    plt.show()
    print "False negative examples:"
    f,ax = plt.subplots(1,len(examples_false_negative), figsize=(75,75))
    for i in range(len(examples_false_negative)):
        ax[i].axes.get_xaxis().set_visible(False)
        ax[i].axes.get_yaxis().set_visible(False)
        ax[i].imshow(examples_false_negative[i], cmap='gray')
    plt.show()
